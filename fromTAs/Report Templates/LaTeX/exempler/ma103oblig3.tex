\section*{Matrise 1}


\begin{align*}
A =
\begin{bmatrix}
2 & 1 \\
-1 & 2
\end{bmatrix}
\end{align*}

\paragraph{Egenverdiene:} Vi finner først det karakteristiske polynomet.

\begin{multline*}
C_x (A) = \det (xI -A) = \det
\begin{bmatrix}
x-2 &		 -1 \\
1 &		x-2	
\end{bmatrix}
= (x-2)^2 +1 = x^2 -4x + 5 
\end{multline*}
Vi faktoriserer dette utrykket ved å løse $x^2-4x+5=0$
\begin{align*}
x=\frac{4\pm\sqrt{(-4)^2-4\cdot1\cdot5}}{2}=\frac{4\pm\sqrt{-4}}{2}
=\frac{4\pm\sqrt{4(i)^2}}{2}
=\frac{4\pm2i}{2}
=2\pm i
\end{align*}
Vi får da at
\begin{align*}
C_x(A) = \bigl( x-(2+i) \bigr) \bigl( x-(2-i) \bigr)
\end{align*}


Egenverdiene til $A$ er da de komplekse tallene $\lambda_1 = 2+i$ og
$\lambda_2 = 2-i$. Vi finner så matrisens egenvektorer.


\paragraph{Egenrommet $E_{2-i}$:}
Vi er her ute etter $X$ som løser ligningen $\bigl( (2-i)I - A \bigr)X
= 0$. Dette gjøres vha. gaussisk eliminasjon:

\begin{multline*}
%
\left[ \begin{array}{cc|c}
(2-i)-2 &	-1 &	0 \\
1 &	(2-i)-2 &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
-i &	-1 &	0 \\
1 &	-i &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
1 &	-i &	0 \\
1 &	-i &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
1 &	-i &	0 \\
0 &	0 &	0
\end{array} \right] \\
\end{multline*}
%
Som gir oss ligningene
%
\begin{align*}
x_2 & = s \\
x_1 - i x_2 = 0 \Leftrightarrow x_1 & = i s \\
\end{align*}
%
og vi får løsningen $X = s X_1$, hvor $X_1 = 
\begin{bmatrix}
i \\ 1
\end{bmatrix}$, og følgelig $E_{2-i} = \Span \left\{ X_1 \right\} $.





\paragraph{Egenrommet $E_{2+i}$:}

Vi løser ligningen $\bigl( (2+i)I - A \bigr)X = 0$ vha gaussisk
eliminasjon:
%
\begin{multline*}
%
\left[ \begin{array}{cc|c}
(2+i)-2 &	-1 &	0 \\
1 &	(2+i)-2 &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
i &	-1 &	0 \\
1 &	i &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
-1 &	-i &	0 \\
1 &	-i &	0
\end{array} \right] 
%
\rightarrow
%
\left[ \begin{array}{cc|c}
1 &	i &	0 \\
0 &	0 &	0
\end{array} \right] \\
\end{multline*}
%
%
Som gir oss ligningene
%
\begin{align*}
x_2 & = s \\
x_1 + i x_2 = 0 \Leftrightarrow x_1 & = - i s \\
\end{align*}
og vi får løsningen $X = s X_2$, hvor $X_2 = \overline{X_1} =  
\begin{bmatrix}
-i \\ 1
\end{bmatrix}$, og følgelig $E_{2+i} = \Span \left\{ X_2 \right\}$.



\paragraph{Diagonaliseringsmatrisen $P$:}

Egenvektorene er i følge theorem 2\S6.2 linært uavhengie, og $P$ er
derfor gitt ved $P=
\begin{bmatrix} X_1 & X_2 \end{bmatrix} =
\begin{bmatrix}i & -i \\ 1 & 1\end{bmatrix}$. 


\paragraph{Den inverse $P^{-1}$:} 
\begin{multline*}
\left[ \begin{array}{c|c}
P & I
\end{array} \right]
=
\left[ \begin{array}{cc|cc}
i & -i & 1 & 0\\
1 & 1 & 0 & 1
\end{array}\right]
\rightarrow
\left[ \begin{array}{cc|cc}
-1 & 1 & i & 0 \\
1 & 1 & 0 & 1
\end{array}\right]
\rightarrow
\left[ \begin{array}{cc|cc}
1 & 1 & 0 & 1 \\
0 & 2 & i & 1
\end{array}\right]
\rightarrow \\
\left[ \begin{array}{cc|cc}
1 & 1 & 0 & 1 \\
0 & 1 & \frac{i}{2} & \frac{1}{2}
\end{array}\right]
\rightarrow 
\left[ \begin{array}{cc|cc}
1 & 0 & -\frac{i}{2} & \frac{1}{2} \\
0 & 1 & \frac{i}{2} & \frac{1}{2}
\end{array}\right]
=
\left[ \begin{array}{c|c}
I & P^{-1}
\end{array} \right]
\end{multline*}
Altså har vi at $P^{-1}=\begin{bmatrix}
-\frac{i}{2} & \frac{1}{2} \\
\frac{i}{2} & \frac{1}{2}
\end{bmatrix}
$.


\paragraph{Diagonal matrisen $D$:}
\begin{multline*}
D = P^{-1} A P = 
\begin{bmatrix}
-\frac{i}{2} & \frac{1}{2} \\
\frac{i}{2} & \frac{1}{2}
\end{bmatrix}
\begin{bmatrix}
2 & 1 \\
-1 & 2
\end{bmatrix}
\begin{bmatrix}
i & -i \\
1 & 1
\end{bmatrix}
=
\begin{bmatrix}
-(\frac{1}{2}+i)	&	1-\frac{i}{2}  \\
-\frac{1}{2}+i		&	1+\frac{i}{2}
\end{bmatrix}
\begin{bmatrix}
i & -i \\
1 & 1
\end{bmatrix}
= \\
\begin{bmatrix}
-(\frac{1}{2}+i)i + (1-\frac{i}{2}) & (\frac{1}{2}+i)i +(1-\frac{i}{2}) \\
(-\frac{1}{2}+i)i + (1+\frac{i}{2}) & (\frac{1}{2}+i)i + (1+\frac{i}{2}) 
\end{bmatrix}
=
\svar{
\begin{bmatrix}
2-i & 0 \\
0 & 2+1
\end{bmatrix}}
\end{multline*}

\begin{align*}

\paragraph{Egenverdiene:} Vi finner først det karakteristiske polynomet.

\begin{multline*}
C_x (A) = \det (xI -A) = \det
\begin{bmatrix}
x-1+5i &	-6 \\
-3 &		x-1-4i	
\end{bmatrix}
= (x-1+5i)(x-1-4i)-18 \\
= x^2 - x - 4xi - x + 1 + 4i + 5xi - 5i + 20 -18 
= x^2 - (2-i)x + 3-i \\
\end{multline*}
Vi faktoriserer dette utrykket ved å løse $x^2-(2-i)x+3-i = 0$

=\frac{(2-i)\pm\sqrt{-9}}{2} 
=\frac{(2-i)\pm3i}{2}

\begin{cases}
1-2i\\
1+i
\end{cases}
\end{multline*}
Vi får dermed at
\begin{align*}
C_x(A)=\bigl(x-(1+i)\bigr)  \bigl(x-(1-2i)\bigr)
\end{align*}

Egenverdiene til $A$ er da de komplekse tallene $\lambda_1 = 1+i$ og

\end{multline*}
Som gir oss ligningene
\begin{align*}
x_2 &= s \\
ix_1 -x_2 = 0 \Leftrightarrow x_1 = \frac{s}{i} &= - s\cdot i
\end{align*}
og vi får løsningen $X = sX_1$, hvor $X_1 = i
\begin{bmatrix}
-i \\ 1
\end{bmatrix}
=
\begin{bmatrix}
1 \\ i
\end{bmatrix}
$, og  $E_{1+i} = \Span \left\{ X_1 \right\}$.



\paragraph{Egenrommet $E_{1-2i}$:}
