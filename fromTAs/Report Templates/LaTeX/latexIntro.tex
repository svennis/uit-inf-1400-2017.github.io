\begin{slide}
\section*{Hva er \LaTeX{}?}

\begin{itemize} 
\item ``Typesettning'' system spesielt egnet for produksjon av
vitenskapelige dokumenter.
\item Skribenten spesifiserer struktur, \LaTeX{} lager layout og
design.
\item Basert på \TeX{} motoren av Donald~E.~Knuth fra 1982.
\item Originalt skrevet av Leslie~Lamport og ble sist oppdatert i
1994. 
\end{itemize}
\end{slide}

\begin{slide}
\section*{Hvorfor bruke \LaTeX}

\begin{itemize}
\item Er gratis!
\item Er støttet på stort sett alt av HW/SW plattformer.
\item Håndterer lett \emph{store} dokumenter.
\item Dokumentene er rene tekstfiler.
\item Profesjonelt resultat.
\item Lar skribenten fokusere på innhold og struktur, og ikke på
utseende.
\end{itemize}
%\end{slide}


%\begin{slide}
\subsection*{Hvorfor ikke bruke \LaTeX}
\begin{itemize}
\item Vanskelig å lære (?).
\item Ingen fancy GUI.
\item Mye å huske.
\item Er ikke et desktop publishing program.
\end{itemize}
\end{slide}





\begin{slide}
\section*{Hvordan fungerer \LaTeX}
\begin{itemize}
\item Skriver dokumentet i en vanlig tekst editor (f.~eks xemacs):
\begin{verbatim}
xemacs dokument.tex
\end{verbatim}
\item Kjøre dokumentet gjennom \LaTeX{} programmet:
\begin{verbatim}
latex dokument.tex 
\end{verbatim}
\item Får så ut en \emph{dvi} fil som man kan:
	\begin{itemize}
	\item Se på ved hjelp av xdvi: \verb|xdvi dokument.dvi|
	\item Skrive ut: \verb|dvips dokument.dvi|
	\item Konvertere til pdf: \verb|dvipdf dokument.dvi|
	\item osv...
	\end{itemize}
\end{itemize}
\end{slide}

\begin{slide}
\section*{Strukturerings elementer}
\begin{itemize}

\item Kommandoer
	\begin{itemize}
	\item Har formen \verb|\command[opt]{text}|.
	\end{itemize}

\item Environment/Miljø
	\begin{itemize}
	\item Begynner med \verb|\begin{navn}|.
	\item Slutter med \verb|\end{navn}|.
	\item Tekst i et miljø har spesielle egenskaper.
	\item Definerer spesielle kommandoer.
	\end{itemize}

\item Dokument klasse \verb|\documentclass[opts]{klasse}|
	\begin{itemize}
	\item Spesifiserer type dokument du ønsker å skrive. For eksempel en artikkel (article), en book (book), en raport (report), et brev (letter), osv.
	\end{itemize}
\item Finnes et utall kommandoer og miljøer, her presenteres bare et
lite utvalg. 
\item Man kan lett lage sine egne kommandoer og miljøer.
\end{itemize}
\end{slide}

\begin{slide}
\section*{Å skrive med \LaTeX}

Å skrive \LaTeX tekst foregår litt forskjellig i forhold til
tekstbehandlings programmer:

\begin{itemize}

\item Tabulator, mellomrom og linjeskift ignoreres:
{\small
\begin{demo}{Eksempel}
\LaTeX is a document
     preparation 
system       designed  by Leslie 
Lamport      in  1985.
\end{demo}
}

\item To linjeskift gir angir start på en ny paragraf.

\item Har 10 spesialtegn som ikke kan brukes direkte i teksten, disse
er:

{ \small
\begin{demo}{Spesialtegn}
$\backslash$  $\sim$  \# \$ 
 \%  \&  \_  \{  \}  $\hat{}$
\end{demo}
}
\end{itemize}

\begin{slide}
\section*{Å skrive med \LaTeX}

\begin{itemize}

\item Siteringstegn er ikke \verb|"| , men man bruker doble \verb|`| og \verb|'| tegn. Altså
ikke {\center{ {\huge \verb|"sitat"|}, men  {\huge  ``\verb|sitat|'' }}}.

\item Alt etter et \% tegn behandles som kommentarer og vises ikke i
dokumentet. 
{\small
\begin{demo}{Eksempel}
  Dette er vanlig tekst\\
  % Dette er en kommentar
  Men % alt dette vises ikke.
  vises dette? 

\end{demo}
}

\end{itemize}

\end{slide}










\begin{slide}
\section*{Seksjonerings kommandoer}
\begin{itemize}
\item Lager overskrifter på forskjellig nivå
	\begin{list}{}{}
	\item \verb|\part{}|
	\item \verb|\chapter{}| 
	\item \verb|\section{}|
	\item \verb|\subsection{}|
	\item \verb|\subsubsection{}|  
	\item \verb|\paragraph{}|
	\item \verb|\subparagraph{}|
	\end{list}
\item Kommandoen \verb|\section{Hei å hå!}| lager en overskrift med teksten
``Hei å hå!''.
\item Seksjonene er nummererte. Man kan få unummererte seksjoner ved å
bruke * etter seksjonskommandoen. F.eks \verb|\section*{hei å hå!}|

\end{itemize}

\end{slide}

\begin{slide}
\section*{Verbatim}
\begin{itemize}
\item Inkluderer tekst akkurat som den er skrevet.
\item \verb|\begin{verbatim} ... \end{verbatim}| for tekst som går
over flere linjer.
\item \verb§\verb|tekst|§ for ord.

\item \verb|\verbatiminput{fil}| for å inkludere en hel fil.
\end{itemize}
{\small
\begin{demo}{Eksempel}
\begin{verbatim}
for (i=0; i<10; i++){
   j = j + 1;
}
\end{verbatim}
\end{demo}
}
\end{slide}

\section*{Tabeller}


\begin{itemize}


\item \verb|\begin{tabular}{def}...\end{tabular}|

\begin{slide}
\section*{Figurer og floats}
\begin{itemize}
\item Tabeller og figurer lar man gjerne flyte i teksten.
	\begin{itemize}
	\item \verb|\begin{figure} ... \end{figure}| for figurer
	\item \verb|\begin{table} ... \end{table}| for tabeller.
	\end{itemize}
\item \verb|\caption{...}| lager ``billedtekst'' til floaten.
\item Lager referanse i teksten (feks. ``dette er vist i Figur 1.''). 
	\begin{itemize}
	\item \verb|\label{ns:tekst}| lager en referanse til floaten (ns er navnerom).
	\item \verb|\ref{ns:tekst}| refererer til floaten
	\end{itemize}
\end{itemize}
%
{\small
\begin{demo}{Eksempel}
\begin{figure}
\includegraphics[width=5cm]{tdlogo}
\caption{TD sin logo}
\label{fig:tdlogo}
\end{figure}
... se Figur \ref{fig:tdlogo}. 
\end{demo}
}
\end{slide}
